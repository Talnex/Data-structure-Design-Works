class node{
public:
    int path[NUM_POINTS];
    node * next;
    node(){
        for (int i=0; i<NUM_POINTS; i++) {
            path[i]=-1;
        }
        next=NULL;
    }
};
node* node_init(){
    node *head=new node;
    return head;
}
node * insert(node* head,int * path){
    node *a=new node;
    head->next=a;
    for (int i=0; path[i]!=-1; i++) {
        a->path[i]=path[i];
    }
    return head;
}

class stack{
public:
    int num[NUM_POINTS];
    int top;
    stack(){
        for (int i=0; i<NUM_POINTS; i++) {
            num[i]=-1;
        }
        top=0;
    }
    int read(){
        return num[top-1];
    }
    int pop(){
        return num[--top];
    }
    void push(int a){
        num[top++]=a;
    }
    bool is_empty(){
        return top?0:1;
    }
    bool do_have(int a){
        for (int i=0; i<top; i++) {
            if (num[i]==a) return 1;
        }
        return 0;
    }
};
